# NayePankh Foundation — Python Automation Pipeline
# Volunteer Data Processing & Report Generation

import json, csv, datetime
from collections import defaultdict

class NayePankhAutomation:
    def __init__(self):
        self.volunteers = []
        self.campaigns  = []
    
    def load_volunteer_data(self, filepath):
        """Load and validate volunteer CSV data"""
        with open(filepath) as f:
            reader = csv.DictReader(f)
            for row in reader:
                if self._validate(row):
                    self.volunteers.append(row)
        return len(self.volunteers)
    
    def generate_weekly_report(self):
        """Auto-generate impact summary"""
        cities = defaultdict(int)
        skills = defaultdict(int)
        for v in self.volunteers:
            cities[v['city']] += 1
            for s in v['skills'].split(','):
                skills[s.strip()] += 1
        return {
            "total": len(self.volunteers),
            "cities": dict(cities),
            "top_skill": max(skills, key=skills.get),
            "generated": datetime.date.today().isoformat()
        }

bot = NayePankhAutomation()
bot.load_volunteer_data("volunteers.csv")
report = bot.generate_weekly_report()
print(json.dumps(report, indent=2))
